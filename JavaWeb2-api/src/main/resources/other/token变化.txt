验证码可以由前端校验，后端获取解码规则进行校验

public class Test {
	
	public static final String PUBLIC_KEY = "1A@4#3D%5q$2v!";
	
	public static void main(String[] args) throws Exception {
		String getDate = "20180815151711";
		System.out.println("原字符串为:"+PUBLIC_KEY);
		String tempArray[] = new String[PUBLIC_KEY.length()];
		for(int i=0;i<tempArray.length;i++){
			tempArray[i] = String.valueOf(PUBLIC_KEY.charAt(i));
		}
		for(int i=0;i<getDate.length();i++){
			String str = String.valueOf(getDate.charAt(i));//获得日期字符串的每位数字
			int m = i;
			int n = Integer.parseInt(str);
			str = tempArray[m];
			tempArray[m] = tempArray[n];
			tempArray[n] = str;
		}
		StringBuilder sb = new StringBuilder();
		for(int i=0;i<tempArray.length;i++){
			sb.append(tempArray[i]);
		}
		System.out.println("现字符串为:"+sb.toString());
	}

}
